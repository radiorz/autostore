const { state,$ } = createStore({    
  root:{
    count: 1
  }    
})

export  const Counter = () => {  
  return (
    <Card title="显示状态值">
      <span>Count:{$('root.count')}</span>
      <Button onClick={()=>state.root.count++}>+1</Button>
    </Card>
  );
};
